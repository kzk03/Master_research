# ============================================================
# サーバ実行手順 v1（初回実行、/mnt/data1/kazuki-h/Master_research）
# ============================================================

# 1. ローカル → サーバに同期
rsync -avz --exclude='.git/' --progress /Users/kazuki-h/Master_research socsel:/mnt/data1/kazuki-h/

# 2. サーバにSSH
ssh socsel

# 3. 作業ディレクトリに移動
cd /mnt/data1/kazuki-h/Master_research

# 4. 軌跡キャッシュが既にあるか確認（あればスキップされる）
ls -lh outputs/variant_comparison_server/trajectory_cache/*.pkl 2>/dev/null

# 5. 全パイプライン実行（軌跡抽出→学習→評価、non-MT 3バリアント）
nohup bash scripts/run_all.sh outputs/variant_comparison_server \
    > outputs/variant_comparison_server/logs/main.log 2>&1 &

# 6. 進捗確認
tail -f outputs/variant_comparison_server/logs/main.log
tail -1 outputs/variant_comparison_server/logs/*.log

# 7. 完了後、結果をローカルに同期（ローカルで実行）
rsync -avz --progress socsel:/mnt/data1/kazuki-h/Master_research/outputs/ ./outputs/
